#include <iostream>
using namespace std;


int main()
{   
    int student_grades[4][3] , sum[3] ={0}, avg[3] ;
    for(int i=0;i<3;i++)//student
    {
        for(int j=0;j<4;j++){//subjects
            cout<<"Enter the grade of subject "<<j+1<<" for student "<<i+1<<" : ";
            cin>>student_grades[j][i];
            sum[i]+=student_grades[j][i];
        }
        avg[i]=sum[i]/4;
    }
    cout<<endl<<"\t\t";
    for(int i=0;i<3;i++){
        cout<<"student "<< i+1<<"\t\t";
    }
    cout<<endl;
    for(int i=0;i<4;i++)
    {
        cout<<"\t\t";
        for(int j=0;j<3;j++){
            cout<<student_grades[i][j]<<"\t\t\t";
        }
        cout<<endl;
    }
    cout<<"---------------------------------------------------------------------"<<endl;
    cout<<"Sum"<<"\t\t";
    for(int i=0;i<3;i++){
        cout<<sum[i]<<"\t\t\t";
    }
    cout<<endl<<"Average"<<"\t\t";
    for(int i=0;i<3;i++){
        cout<<avg[i]<<"\t\t\t";
    }
    return 0;
}
